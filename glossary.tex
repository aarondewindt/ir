\newglossaryentry{x_k}{
name = $\bar x_k $,
description =  Current real system state,
}


\newglossaryentry{x_k1}{
name = $\bar x_{k+1}$,
description =  One-step-ahead real system state ,
}

\newglossaryentry{x_kk}{
name = $\hat {\bar x}_{k,k}$,
description =  Current optimal estimated system state ,
}

\newglossaryentry{x_k1k}{
name = $\hat {\bar x}_{k+1, k}$,
description =  One-step-ahead state prediction,
}


\newglossaryentry{x_k1k1}{
name = $\hat {\bar x}_{k+1, k+1}$,
description =  One-step-ahead optimal estimated system state,
}

\newglossaryentry{eps_k1k}{
name = $\hat {\bar \epsilon}_{k+1, k}$,
description =  State prediction error. $[\hat {\bar x}_{k+1, k} - \bar x_{k+1}]$,
}
